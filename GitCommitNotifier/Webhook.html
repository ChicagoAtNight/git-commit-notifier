def payload
  pay = {
    'repository' => {
      'name' => @repo_name
    },
    'ref' => @ref_name,
    'before' => @old_rev,
    'after' => @new_rev,
    'commits' => [
      {
        'added' => @changed[:a],
        'modified' => @changed[:m],
        'removed' => @changed[:d],
        'committer' => {
          'name' => @committer,
          'email' => @email
        },
        'message' => CGI::escape(@message)
      }
    ]
  }
  Yajl::Encoder.encode(pay)
end
